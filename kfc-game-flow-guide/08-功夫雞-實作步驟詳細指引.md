### Step 1.1 — RoundStatus Value Object

**檔案**：`gameservice/internal/domain/entity/round_status.go`

```go
package entity

// RoundStatus 代表多人共享回合的生命週期狀態。
type RoundStatus string

const (
	// RoundStatusCreated 已建立：管理員建立回合，尚未開放下注。
	RoundStatusCreated RoundStatus = "created"
	// RoundStatusOpen 開盤中：玩家可下注。
	RoundStatusOpen RoundStatus = "open"
	// RoundStatusClosed 封盤：不再接受下注，等待結果。
	RoundStatusClosed RoundStatus = "closed"
	// RoundStatusSettled 結算中：結果已確定，系統正在批次派彩。
	RoundStatusSettled RoundStatus = "settled"
	// RoundStatusCompleted 已完成：所有派彩完成，回合歸檔。
	RoundStatusCompleted RoundStatus = "completed"
	// RoundStatusCancelled 已取消：管理員取消回合，全額退款。
	RoundStatusCancelled RoundStatus = "cancelled"
)

// IsTerminal 回傳此狀態是否為終態（不可再轉換）。
func (s RoundStatus) IsTerminal() bool {
	return s == RoundStatusCompleted || s == RoundStatusCancelled
}

// CanTransitionTo 檢查是否允許從當前狀態轉換到目標狀態。
func (s RoundStatus) CanTransitionTo(target RoundStatus) bool {
	switch s {
	case RoundStatusCreated:
		return target == RoundStatusOpen
	case RoundStatusOpen:
		return target == RoundStatusClosed || target == RoundStatusCancelled
	case RoundStatusClosed:
		return target == RoundStatusSettled || target == RoundStatusCancelled
	case RoundStatusSettled:
		return target == RoundStatusCompleted
	default:
		return false
	}
}
```

package entity

// BetOption 代表功夫雞遊戲的投注選項。
type BetOption string

const (
	// BetOptionWala 藍方 (Wala)。
	BetOptionWala BetOption = "wala"
	// BetOptionMelo 紅方 (Melo)。
	BetOptionMelo BetOption = "melo"
	// BetOptionDraw 平手 (Draw)。
	BetOptionDraw BetOption = "draw"
)

// ValidBetOptions 所有合法的投注選項。
var ValidBetOptions = []BetOption{BetOptionWala, BetOptionMelo, BetOptionDraw}

// IsValid 驗證投注選項是否合法。
func (o BetOption) IsValid() bool {
	for _, v := range ValidBetOptions {
		if o == v {
			return true
		}
	}
	return false
}

package entity

import "time"

// Round 代表一個多人共享的遊戲回合（Aggregate Root）。
// 純資料型別，狀態轉換由 UseCase 負責。
type Round struct {
	RoundID     string
	GameID      string // 遊戲 ID，例 "kungfuchicken"
	Status      RoundStatus
	Result      *BetOption // nil = 尚未開獎
	HouseEdge   float64    // 抽水比例，例 0.05 = 5%
	BetPool     BetPool    // 各選項投注總額
	TotalPool   float64    // 投注池總額
	BetCount    int        // 總下注筆數
	CreatedBy   string     // 建立者（Admin ID）
	OpenedAt    *time.Time
	ClosedAt    *time.Time
	SettledAt   *time.Time
	CompletedAt *time.Time
	CancelledAt *time.Time
	CreatedAt   time.Time
}

// BetPool 各投注選項的金額池。
type BetPool struct {
	Wala float64
	Melo float64
	Draw float64
}

// Total 回傳投注池總額。
func (p BetPool) Total() float64 {
	return p.Wala + p.Melo + p.Draw
}

// GetByOption 依投注選項取得該選項的投注總額。
func (p BetPool) GetByOption(option BetOption) float64 {
	switch option {
	case BetOptionWala:
		return p.Wala
	case BetOptionMelo:
		return p.Melo
	case BetOptionDraw:
		return p.Draw
	default:
		return 0
	}
}

// RoundOdds 即時賠率計算結果（Value Object，非持久化）。
type RoundOdds struct {
	WalaOdds  float64
	MeloOdds  float64
	DrawOdds  float64
	TotalPool float64
	WalaPool  float64
	MeloPool  float64
	DrawPool  float64
}

// GetOddsByOption 依投注選項取得該選項的賠率。
func (o RoundOdds) GetOddsByOption(option BetOption) float64 {
	switch option {
	case BetOptionWala:
		return o.WalaOdds
	case BetOptionMelo:
		return o.MeloOdds
	case BetOptionDraw:
		return o.DrawOdds
	default:
		return 0
	}
}

**檔案**：`gameservice/internal/usecase/kfc_odds/odds.go`

```go
package kfc_odds

import "github.com/game-pipeline/gameservice/internal/domain/entity"

type calculator struct{}

// New 建立賠率計算器。
func New() Calculator {
	return &calculator{}
}

// Calculate 依據投注池與抽水比例計算各選項的 Parimutuel 賠率。
//
// 公式：odds(outcome) = totalPool × (1 - houseEdge) / outcomePool
func (c *calculator) Calculate(pool *entity.BetPool, houseEdge float64) *entity.RoundOdds {
	total := pool.Total()
	if total == 0 {
		return &entity.RoundOdds{}
	}
	effective := total * (1 - houseEdge)
	return &entity.RoundOdds{
		WalaOdds:  safeDiv(effective, pool.Wala),
		MeloOdds:  safeDiv(effective, pool.Melo),
		DrawOdds:  safeDiv(effective, pool.Draw),
		TotalPool: total,
		WalaPool:  pool.Wala,
		MeloPool:  pool.Melo,
		DrawPool:  pool.Draw,
	}
}

// CalculatePayout 計算單筆投注的派彩金額。
func (c *calculator) CalculatePayout(betAmount, odds float64) float64 {
	return betAmount * odds
}

func safeDiv(numerator, denominator float64) float64 {
	if denominator == 0 {
		return 0
	}
	return numerator / denominator
}
```

package kfc_odds

import (
	"testing"

	"github.com/game-pipeline/gameservice/internal/domain/entity"
)

func TestCalculate_NormalDistribution(t *testing.T) {
	calc := New()
	pool := &entity.BetPool{Wala: 6000, Melo: 3000, Draw: 1000}
	odds := calc.Calculate(pool, 0.05)

	// totalPool=10000, effective=9500
	// walaOdds = 9500/6000 ≈ 1.5833
	assertApprox(t, "WalaOdds", odds.WalaOdds, 1.5833, 0.001)
	assertApprox(t, "MeloOdds", odds.MeloOdds, 3.1667, 0.001)
	assertApprox(t, "DrawOdds", odds.DrawOdds, 9.5000, 0.001)
	assertApprox(t, "TotalPool", odds.TotalPool, 10000, 0.001)
}

func TestCalculate_EmptyPool(t *testing.T) {
	calc := New()
	pool := &entity.BetPool{}
	odds := calc.Calculate(pool, 0.05)

	if odds.WalaOdds != 0 || odds.MeloOdds != 0 || odds.DrawOdds != 0 {
		t.Errorf("empty pool should yield zero odds, got %+v", odds)
	}
}

func TestCalculate_SingleOptionHasBets(t *testing.T) {
	calc := New()
	pool := &entity.BetPool{Wala: 1000, Melo: 0, Draw: 0}
	odds := calc.Calculate(pool, 0.05)

	// 只有 Wala 有人投注：walaOdds = 1000*0.95/1000 = 0.95
	assertApprox(t, "WalaOdds", odds.WalaOdds, 0.95, 0.001)
	if odds.MeloOdds != 0 {
		t.Errorf("MeloOdds should be 0, got %f", odds.MeloOdds)
	}
	if odds.DrawOdds != 0 {
		t.Errorf("DrawOdds should be 0, got %f", odds.DrawOdds)
	}
}

func TestCalculate_ZeroHouseEdge(t *testing.T) {
	calc := New()
	pool := &entity.BetPool{Wala: 5000, Melo: 5000, Draw: 0}
	odds := calc.Calculate(pool, 0)

	// houseEdge=0: walaOdds = 10000/5000 = 2.0
	assertApprox(t, "WalaOdds", odds.WalaOdds, 2.0, 0.001)
	assertApprox(t, "MeloOdds", odds.MeloOdds, 2.0, 0.001)
}

func TestCalculatePayout(t *testing.T) {
	calc := New()
	payout := calc.CalculatePayout(100, 1.5833)
	assertApprox(t, "Payout", payout, 158.33, 0.01)
}

func assertApprox(t *testing.T, name string, got, want, tolerance float64) {
	t.Helper()
	diff := got - want
	if diff < 0 {
		diff = -diff
	}
	if diff > tolerance {
		t.Errorf("%s: got %.4f, want %.4f (tolerance %.4f)", name, got, want, tolerance)
	}
}

package kfc_round_mgmt

import (
	"context"
	"fmt"
	"time"

	"github.com/game-pipeline/gameservice/internal/domain/entity"
	"github.com/game-pipeline/gameservice/internal/domain/port"
	"github.com/game-pipeline/gameservice/internal/usecase/kfc_round_mgmt/ports"
	"github.com/google/uuid"
)

type createUseCase struct {
	roundRepo  ports.RoundRepository
	stateStore port.RoundStateStore
}

// NewCreateUseCase 建立 CreateUseCase。
func NewCreateUseCase(roundRepo ports.RoundRepository, stateStore port.RoundStateStore) CreateUseCase {
	return &createUseCase{roundRepo: roundRepo, stateStore: stateStore}
}

func (uc *createUseCase) Execute(ctx context.Context, cmd *CreateCommand) (*RoundResult, error) {
	if cmd.HouseEdge < 0 || cmd.HouseEdge >= 1 {
		return nil, fmt.Errorf("invalid house edge: %.4f (must be 0 <= x < 1)", cmd.HouseEdge)
	}

	round := &entity.Round{
		RoundID:   uuid.New().String(),
		GameID:    cmd.GameID,
		Status:    entity.RoundStatusCreated,
		HouseEdge: cmd.HouseEdge,
		CreatedBy: cmd.CreatedBy,
		CreatedAt: time.Now(),
	}

	if err := uc.roundRepo.Create(ctx, round); err != nil {
		return nil, fmt.Errorf("create round: %w", err)
	}

	if err := uc.stateStore.InitPool(ctx, round.RoundID); err != nil {
		return nil, fmt.Errorf("init pool: %w", err)
	}

	return &RoundResult{Round: round}, nil
}

package kfc_round_mgmt

import (
	"context"
	"fmt"
	"time"

	"github.com/game-pipeline/gameservice/internal/domain/entity"
	"github.com/game-pipeline/gameservice/internal/usecase/kfc_round_mgmt/ports"
)

type openUseCase struct {
	roundRepo ports.RoundRepository
}

// NewOpenUseCase 建立 OpenUseCase。
func NewOpenUseCase(roundRepo ports.RoundRepository) OpenUseCase {
	return &openUseCase{roundRepo: roundRepo}
}

func (uc *openUseCase) Execute(ctx context.Context, roundID string) (*RoundResult, error) {
	round, err := uc.roundRepo.FindByID(ctx, roundID)
	if err != nil {
		return nil, fmt.Errorf("find round: %w", err)
	}
	if round == nil {
		return nil, fmt.Errorf("round not found: %s", roundID)
	}

	if !round.Status.CanTransitionTo(entity.RoundStatusOpen) {
		return nil, fmt.Errorf("cannot open round in status %s", round.Status)
	}

	now := time.Now()
	updates := map[string]interface{}{"opened_at": now}
	if err := uc.roundRepo.UpdateStatus(ctx, roundID, entity.RoundStatusCreated, entity.RoundStatusOpen, updates); err != nil {
		return nil, fmt.Errorf("update status: %w", err)
	}

	round.Status = entity.RoundStatusOpen
	round.OpenedAt = &now
	return &RoundResult{Round: round}, nil
}

		wallet:       wallet,
		configCache:  configCache,
		stateStore:   stateStore,
		outboxWriter: outboxWriter,
	}
}

func (uc *cancelUseCase) Execute(ctx context.Context, roundID string) (*RoundResult, error) {
	round, err := uc.roundRepo.FindByID(ctx, roundID)
	if err != nil {
		return nil, fmt.Errorf("find round: %w", err)
	}
	if round == nil {
		return nil, fmt.Errorf("round not found: %s", roundID)
	}

	if round.Status != entity.RoundStatusOpen && round.Status != entity.RoundStatusClosed {
		return nil, fmt.Errorf("cannot cancel round in status %s", round.Status)
	}

	now := time.Now()
	updates := map[string]interface{}{"cancelled_at": now}
	if err := uc.roundRepo.UpdateStatus(ctx, roundID, round.Status, entity.RoundStatusCancelled, updates); err != nil {
		return nil, fmt.Errorf("update status: %w", err)
	}

	// 全額退款所有已下注玩家
	bets, err := uc.betRepo.FindConfirmedByRoundID(ctx, roundID)
	if err != nil {
		slog.ErrorContext(ctx, "failed to load bets for cancel", "roundID", roundID, "error", err)
	} else {
		for _, bet := range bets {
			uc.refundBet(ctx, bet)
		}
	}

	// 清理 Redis 投注池
	if delErr := uc.stateStore.DeletePool(ctx, roundID); delErr != nil {
		slog.WarnContext(ctx, "failed to delete pool", "roundID", roundID, "error", delErr)
	}

	round.Status = entity.RoundStatusCancelled
	round.CancelledAt = &now
	return &RoundResult{Round: round}, nil
}

func (uc *cancelUseCase) refundBet(ctx context.Context, bet *entity.KFCBet) {
	integrator, ok := uc.configCache.GetIntegrator(bet.IID)
	if !ok {
		slog.ErrorContext(ctx, "integrator not found for refund", "iid", bet.IID, "betID", bet.BetID)
		return
	}

	cancelTxID := uuid.New().String()
	_, err := uc.wallet.Cancel(ctx, integrator.WalletEndpoints.Cancel, &port.CancelRequest{
		UserID:       bet.UserID,
		TxID:         cancelTxID,
		OriginalTxID: bet.DebitTxID,
	})
	if err != nil {
		slog.ErrorContext(ctx, "cancel debit failed, writing to outbox",
			"betID", bet.BetID, "roundID", bet.RoundID, "error", err)
		outboxCtx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
		defer cancel()
		_ = uc.outboxWriter.Write(outboxCtx, outbox.EventTypeCancelDebit, map[string]interface{}{
			"cancelEndpoint": integrator.WalletEndpoints.Cancel,
			"userId":         bet.UserID,
			"roundId":        bet.RoundID,
			"betId":          bet.BetID,
			"debitTxId":      bet.DebitTxID,
			"cancelTxId":     cancelTxID,
			"betAmount":      bet.BetAmount,
		})
	}
}

func (uc *useCase) Execute(ctx context.Context, cmd *Command) (*Result, error) {
	slog.DebugContext(ctx, "kfc.placeBet",
		"roundID", cmd.RoundID, "userID", cmd.UserID,
		"option", cmd.Option, "betAmount", cmd.BetAmount,
	)

	// ── Step 1：驗證 ────────────────────────────────────
	if !cmd.Option.IsValid() {
		return nil, fmt.Errorf("invalid bet option: %s", cmd.Option)
	}

	round, err := uc.roundReader.FindByID(ctx, cmd.RoundID)
	if err != nil {
		return nil, fmt.Errorf("find round: %w", err)
	}
	if round == nil {
		return nil, fmt.Errorf("round not found: %s", cmd.RoundID)
	}
	if round.Status != entity.RoundStatusOpen {
		return nil, fmt.Errorf("round is not open (current: %s)", round.Status)
	}

	// 檢查同一玩家同一回合同一選項是否已下注
	exists, err := uc.betRepo.ExistsByUserRoundOption(ctx, cmd.UserID, cmd.RoundID, cmd.Option)
	if err != nil {
		return nil, fmt.Errorf("check duplicate bet: %w", err)
	}
	if exists {
		return nil, fmt.Errorf("duplicate bet: user %s already bet %s on round %s", cmd.UserID, cmd.Option, cmd.RoundID)
	}

	// ── Step 2：取得 Integrator endpoints ────────────────
	integrator, ok := uc.configCache.GetIntegrator(cmd.IID)
	if !ok {
		return nil, fmt.Errorf("integrator not found: %s", cmd.IID)
	}

	// ── Step 3：Wallet Debit ────────────────────────────
	ctx, spanDebit := tracing.Start(ctx, "kfc.placeBet.debitWallet")
	betID := uuid.New().String()
	debitTxID := fmt.Sprintf("kfc-%s-debit", betID)

	debitCtx, debitCancel := context.WithTimeout(ctx, walletTimeout)
	debitResult, err := uc.wallet.Debit(debitCtx, integrator.WalletEndpoints.Debit, &port.DebitRequest{
		UserID:      cmd.UserID,
		Amount:      cmd.BetAmount,
		TxID:        debitTxID,
		GameID:      cmd.GameID,
		RoundID:     cmd.RoundID,
		PlayerToken: cmd.PlayerToken,
	})
	debitCancel()
	spanDebit.End()
	if err != nil {
		return nil, fmt.Errorf("debit failed: %w", err)
	}

	// ── Step 4：Redis 原子更新投注池 ─────────────────────
	ctx, spanPool := tracing.Start(ctx, "kfc.placeBet.updatePool")
	if err := uc.stateStore.IncrementPool(ctx, cmd.RoundID, cmd.Option, cmd.BetAmount); err != nil {
		slog.ErrorContext(ctx, "CRITICAL: pool increment failed after debit, cancelling",
			"betID", betID, "roundID", cmd.RoundID, "error", err)
		spanPool.End()
		uc.cancelDebit(ctx, integrator.WalletEndpoints.Cancel, cmd.UserID, debitTxID)
		return nil, fmt.Errorf("pool update failed: %w", err)
	}
	spanPool.End()

	// ── Step 5：Double Check Round 狀態 ─────────────────
	round2, err := uc.roundReader.FindByID(ctx, cmd.RoundID)
	if err != nil || round2 == nil || round2.Status != entity.RoundStatusOpen {
		slog.WarnContext(ctx, "round closed during bet, rolling back",
			"betID", betID, "roundID", cmd.RoundID)
		_ = uc.stateStore.DecrementPool(ctx, cmd.RoundID, cmd.Option, cmd.BetAmount)
		uc.cancelDebit(ctx, integrator.WalletEndpoints.Cancel, cmd.UserID, debitTxID)
		return nil, fmt.Errorf("round closed during bet placement")
	}

	// ── Step 6：計算即時賠率 ─────────────────────────────
	pool, err := uc.stateStore.GetPool(ctx, cmd.RoundID)
	if err != nil {
		slog.WarnContext(ctx, "failed to get pool for odds", "error", err)
		pool = &round.BetPool // fallback 用 Round 上的值
	}
	currentOdds := uc.oddsCalc.Calculate(pool, round.HouseEdge)
	betOdds := currentOdds.GetOddsByOption(cmd.Option)

	// ── Step 7：計算 USD 等值 ────────────────────────────
	var betAmountUSD, exchangeRate float64
	if currency, ok := uc.configCache.GetCurrency(cmd.Currency); ok && currency.ExchangeRate > 0 {
		exchangeRate = currency.ExchangeRate
		betAmountUSD = cmd.BetAmount / exchangeRate
	}

	// ── Step 8：MongoDB 寫入 KFCBet ─────────────────────
	bet := &entity.KFCBet{
		BetID:        betID,
		RoundID:      cmd.RoundID,
		UserID:       cmd.UserID,
		GameID:       cmd.GameID,
		IID:          cmd.IID,
		Currency:     cmd.Currency,
		Option:       cmd.Option,
		BetAmount:    cmd.BetAmount,
		BetAmountUSD: betAmountUSD,
		ExchangeRate: exchangeRate,
		Odds:         betOdds,
		Status:       entity.BetStatusConfirmed,
		DebitTxID:    debitTxID,
		IsTest:       cmd.IsBot,
		CreatedAt:    time.Now(),
	}
	if err := uc.betRepo.Save(ctx, bet); err != nil {
		slog.ErrorContext(ctx, "CRITICAL: bet save failed after debit+pool update",
			"betID", betID, "roundID", cmd.RoundID, "error", err)
		// 不回滾——Debit 已成功且 Pool 已更新，MongoDB 寫入失敗可後續補償
	}

	return &Result{
		BetID:       betID,
		RoundID:     cmd.RoundID,
		Option:      cmd.Option,
		BetAmount:   cmd.BetAmount,
		Odds:        betOdds,
		Balance:     debitResult.Balance,
		CurrentOdds: currentOdds,
	}, nil
}

func (uc *useCase) cancelDebit(ctx context.Context, cancelEndpoint, userID, debitTxID string) {

	defer cancel()

	cancelTxID := debitTxID + "-cancel"
	_, err := uc.wallet.Cancel(cancelCtx, cancelEndpoint, &port.CancelRequest{
		UserID:       userID,
		TxID:         cancelTxID,
		OriginalTxID: debitTxID,
	})
	if err != nil {
		slog.ErrorContext(ctx, "cancel debit failed", "debitTxID", debitTxID, "error", err)
	}
}

	"context"

const creditTimeout = 12 * time.Second

type useCase struct {

func (uc *useCase) Execute(ctx context.Context, cmd *Command) (*Result, error) {
	if !cmd.Result.IsValid() {
		return nil, fmt.Errorf("invalid result: %s", cmd.Result)
	}

	// ── Step 1：驗證 Round 狀態 ──────────────────────────
	round, err := uc.roundRepo.FindByID(ctx, cmd.RoundID)
	if err != nil {
		return nil, fmt.Errorf("find round: %w", err)
	}
	if round == nil {
		return nil, fmt.Errorf("round not found: %s", cmd.RoundID)
	}
	if round.Status != entity.RoundStatusClosed {
		return nil, fmt.Errorf("round must be closed to settle (current: %s)", round.Status)
	}

	// ── Step 2：讀取最終投注池 + 計算最終賠率 ─────────────
	ctx, spanOdds := tracing.Start(ctx, "kfc.settle.calculateFinalOdds")
	pool, err := uc.stateStore.GetPool(ctx, cmd.RoundID)
	if err != nil {
		spanOdds.End()
		return nil, fmt.Errorf("get pool: %w", err)
	}
	finalOdds := uc.oddsCalc.Calculate(pool, round.HouseEdge)
	spanOdds.End()

	// ── Step 3：更新 Round 為 Settled ────────────────────
	now := time.Now()
	result := cmd.Result
	updates := map[string]interface{}{
		"settled_at": now,
		"result":     string(result),
		"wala_pool":  pool.Wala,
		"melo_pool":  pool.Melo,
		"draw_pool":  pool.Draw,
		"total_pool": pool.Total(),
	}
	if err := uc.roundRepo.UpdateStatus(ctx, cmd.RoundID, entity.RoundStatusClosed, entity.RoundStatusSettled, updates); err != nil {
		return nil, fmt.Errorf("update round to settled: %w", err)
	}

	// ── Step 4：查詢所有 Bets ────────────────────────────
	bets, err := uc.betSettler.FindByRoundID(ctx, cmd.RoundID)
	if err != nil {
		return nil, fmt.Errorf("load bets: %w", err)
	}

	// ── Step 5：逐筆結算 ────────────────────────────────
	ctx, spanSettle := tracing.Start(ctx, "kfc.settle.processBets")
	var winnersCount, losersCount int
	var totalPayout float64
	winOdds := finalOdds.GetOddsByOption(result)

	for _, bet := range bets {
		if bet.Option == result && winOdds > 0 {
			// 贏家
			winAmount := uc.oddsCalc.CalculatePayout(bet.BetAmount, winOdds)
			uc.creditWinner(ctx, bet, winAmount, winOdds)
			winnersCount++
			totalPayout += winAmount
		} else {
			// 輸家
			settledAt := time.Now()
			if err := uc.betSettler.UpdateSettlement(ctx, bet.BetID, entity.BetStatusSettled, 0, 0, "", settledAt); err != nil {
				slog.ErrorContext(ctx, "failed to settle losing bet", "betID", bet.BetID, "error", err)
			}
			losersCount++
		}
	}
	spanSettle.End()

	// ── Step 6：Round → Completed ───────────────────────
	completedAt := time.Now()
	completedUpdates := map[string]interface{}{
		"completed_at": completedAt,
		"bet_count":    len(bets),
	}
	if err := uc.roundRepo.UpdateStatus(ctx, cmd.RoundID, entity.RoundStatusSettled, entity.RoundStatusCompleted, completedUpdates); err != nil {
		slog.ErrorContext(ctx, "failed to mark round completed (settlement done, will be fixed by worker)",
			"roundID", cmd.RoundID, "error", err)
	}

	// ── Step 7：清理 Redis ───────────────────────────────
	if delErr := uc.stateStore.DeletePool(ctx, cmd.RoundID); delErr != nil {
		slog.WarnContext(ctx, "failed to delete pool", "roundID", cmd.RoundID, "error", delErr)
	}

	round.Status = entity.RoundStatusCompleted
	round.Result = &result
	round.SettledAt = &now
	round.CompletedAt = &completedAt
	round.BetPool = *pool
	round.TotalPool = pool.Total()
	round.BetCount = len(bets)

	return &Result{
		Round:        round,
		WinnersCount: winnersCount,
		LosersCount:  losersCount,
		TotalPayout:  totalPayout,
		FinalOdds:    finalOdds,
	}, nil
}

func (uc *useCase) creditWinner(ctx context.Context, bet *entity.KFCBet, winAmount, finalOdds float64) {
	integrator, ok := uc.configCache.GetIntegrator(bet.IID)
	if !ok {
		slog.ErrorContext(ctx, "integrator not found for credit", "iid", bet.IID, "betID", bet.BetID)
		return
	}

	creditTxID := fmt.Sprintf("kfc-%s-credit", bet.BetID)
	creditCtx, cancel := context.WithTimeout(ctx, creditTimeout)
	_, err := uc.wallet.Credit(creditCtx, integrator.WalletEndpoints.Credit, &port.CreditRequest{
		UserID:  bet.UserID,
		Amount:  winAmount,
		TxID:    creditTxID,
		GameID:  bet.GameID,
		RoundID: bet.RoundID,
	})
	cancel()

	settledAt := time.Now()
	if err != nil {
		slog.ErrorContext(ctx, "credit winner failed, writing to outbox",
			"betID", bet.BetID, "winAmount", winAmount, "error", err)
		outboxCtx, outboxCancel := context.WithTimeout(context.Background(), 5*time.Second)
		defer outboxCancel()
		_ = uc.outboxWriter.Write(outboxCtx, outbox.EventTypeCreditWinner, map[string]interface{}{
			"creditEndpoint": integrator.WalletEndpoints.Credit,
			"userId":         bet.UserID,
			"roundId":        bet.RoundID,
			"betId":          bet.BetID,
			"gameId":         bet.GameID,
			"winAmount":      winAmount,
			"creditTxId":     creditTxID,
		})
		return
	}

	if err := uc.betSettler.UpdateSettlement(ctx, bet.BetID, entity.BetStatusDistributed, finalOdds, winAmount, creditTxID, settledAt); err != nil {
		slog.ErrorContext(ctx, "failed to update bet settlement after credit",
			"betID", bet.BetID, "error", err)
	}
}

func toRoundDocument(e *entity.Round) *roundDocument {
	doc := &roundDocument{
		RoundID:     e.RoundID,
		GameID:      e.GameID,
		Status:      string(e.Status),
		HouseEdge:   e.HouseEdge,
		WalaPool:    e.BetPool.Wala,

package mongo

import (
	"context"
	"fmt"

	"github.com/game-pipeline/gameservice/internal/domain/entity"
	"go.mongodb.org/mongo-driver/v2/bson"
	"go.mongodb.org/mongo-driver/v2/mongo"
)

type roundRepository struct {
	collection *mongo.Collection
}

// NewRoundRepository 建立 Round MongoDB Repository。
func NewRoundRepository(db *mongo.Database) *roundRepository {
	return &roundRepository{collection: db.Collection("kfc_rounds")}
}

func (r *roundRepository) Create(ctx context.Context, round *entity.Round) error {
	_, err := r.collection.InsertOne(ctx, toRoundDocument(round))
	return err
}

func (r *roundRepository) FindByID(ctx context.Context, roundID string) (*entity.Round, error) {
	var doc roundDocument
	err := r.collection.FindOne(ctx, bson.M{"_id": roundID}).Decode(&doc)
	if err == mongo.ErrNoDocuments {
		return nil, nil
	}
	if err != nil {
		return nil, err
	}
	return toRoundEntity(&doc), nil
}

func (r *roundRepository) UpdateStatus(ctx context.Context, roundID string, from, to entity.RoundStatus, updates map[string]interface{}) error {
	setFields := bson.M{"status": string(to)}
	for k, v := range updates {
		setFields[k] = v
	}
	result, err := r.collection.UpdateOne(ctx,
		bson.M{"_id": roundID, "status": string(from)},
		bson.M{"$set": setFields},
	)
	if err != nil {
		return err
	}
	if result.MatchedCount == 0 {
		return fmt.Errorf("round %s status is not %s (CAS failed)", roundID, from)
	}
	return nil
}

func (r *roundRepository) FindByGameAndStatus(ctx context.Context, gameID string, status entity.RoundStatus) ([]*entity.Round, error) {
	cursor, err := r.collection.Find(ctx, bson.M{"game_id": gameID, "status": string(status)})
	if err != nil {
		return nil, err
	}
	defer cursor.Close(ctx)

	var docs []roundDocument
	if err := cursor.All(ctx, &docs); err != nil {
		return nil, err
	}

	rounds := make([]*entity.Round, len(docs))
	for i := range docs {
		rounds[i] = toRoundEntity(&docs[i])
	}
	return rounds, nil
}

func (r *roundRepository) FindActiveRound(ctx context.Context, gameID string) (*entity.Round, error) {
	var doc roundDocument
	err := r.collection.FindOne(ctx, bson.M{
		"game_id": gameID,
		"status":  bson.M{"$in": []string{"open", "closed"}},
	}).Decode(&doc)
	if err == mongo.ErrNoDocuments {
		return nil, nil
	}
	if err != nil {
		return nil, err
	}
	return toRoundEntity(&doc), nil
}
